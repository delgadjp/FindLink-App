rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // Deny all read and write access by default
    match /{allPaths=**} {
      allow read, write: if false;
    }

    // Allow read and write for authenticated users in the 'reports' folder
    match /reports/{allPaths=**} {
      allow read, write: if request.auth != null;
    }

    // Allow public read and authenticated user writes to 'tips-app' folder
    match /tips-app/{imageId} {
      // Allow read access for anyone
      allow read: if true;

      // Allow write access only for authenticated users
      allow write: if request.auth != null && 
                    // Validate file type (allow only images)
                    request.resource.contentType.matches('image/.*') &&
                    // Validate file size to be <= 5MB
                    request.resource.size <= 5 * 1024 * 1024;
    }

    // Allow authenticated users to read and write in the 'missing_persons' folder
    match /missing_persons/{allPaths=**} {
      allow read, write: if request.auth != null;
    }

  }
}
